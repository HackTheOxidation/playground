#+TITLE: Experimenting with source code blocks in Org Mode

* A simple source code block

#+begin_src python
  import sys


  class A:

      def __init__(self):
	  self._a = 1

      def __str__(self):
	  return f"A: _a = {self._a}"

      def fun(self):
	  print(f"I am taking up {sys.getsizeof(self)} bytes in memory.")


  if __name__ == '__main__':
      a = A()
      a.fun()
      print(a)
#+end_src
